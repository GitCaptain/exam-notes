\section{Синтаксис исчисления предикатов}
Язык первого порядка состоит из:
\begin{itemize}
  \item \textbf{Предметные переменные}: $x, y, z, \dots$;
  \item \textbf{Функциональные символы}: каждый $f$ имеет арность $n\ge 0$ (константы — $n=0$);
  \item \textbf{Предикатные символы}: каждый $P$ имеет арность $n\ge 0$ (пропозициональные переменные — $n=0$);
  \item \textbf{Логические связки}: $\neg, \vee, \wedge, \rightarrow$;
  \item \textbf{Кванторы}: $\forall x$, $\exists x$;
  \item \textbf{Вспомогательные символы}: скобки, запятые.
\end{itemize}
\textit{Термы} определяются индуктивно:
\begin{enumerate}
  \item переменная — терм;
  \item если $f$ — $n$-местный функциональный символ и $t_1,\dots,t_n$ — термы, то $f(t_1,\dots,t_n)$ — терм.
\end{enumerate}
\textit{Формулы}:
\begin{enumerate}
  \item если $P$ — $n$-местный предикатный символ и $t_1,\dots,t_n$ — термы, то $P(t_1,\dots,t_n)$ — атомарная формула;
  \item если $\varphi,\psi$ — формулы, то $\neg\varphi$, $\varphi\vee\psi$, $\varphi\wedge\psi$, $\varphi\rightarrow\psi$ — формулы;
  \item если $\varphi$ — формула, а $x$ — переменная, то $\forall x.\varphi$ и $\exists x.\varphi$ — формулы.
\end{enumerate}
\subsection{Свободные и связанные вхождения}
Вхождение переменной в формулу \textit{связано}, если оно находится в области действия квантора по этой переменной; иначе \textit{свободно}. Множество свободных переменных формулы $\varphi$ обозначается $FV(\varphi)$.
\subsection{Подстановка}
$\varphi[x:=\theta]$ — результат замены всех свободных вхождений $x$ на терм $\theta$ с возможным переименованием связанных переменных, чтобы избежать захвата. Терм $\theta$ \textit{свободен для подстановки} вместо $x$ в $\varphi$, если ни одно свободное вхождение переменной из $\theta$ не становится связанным после подстановки.

\section{Семантика}
\textit{Модель (интерпретация)} $\mathcal{M}=\langle D, F, P, E\rangle$:
\begin{itemize}
  \item $D\neq\varnothing$ — предметная область;
  \item $F$ сопоставляет каждому $n$-местному функциональному символу $f$ функцию $F(f) : D^n\to D$;
  \item $P$ сопоставляет каждому $n$-местному предикатному символу $\operatorname{Pr}$ функцию $P(\operatorname{Pr}) : D^n\to\{\text{И},\text{Л}\}$;
  \item $E$ сопоставляет каждой переменной $x$ значение $E(x)\in D$ (оценка переменных).
\end{itemize}
Значение терма $\llbracket t\rrbracket$ определяется индуктивно:
$\llbracket x\rrbracket = E(x)$, $\llbracket f(t_1,\dots,t_n)\rrbracket = F(f)(\llbracket t_1\rrbracket,\dots,\llbracket t_n\rrbracket)$.
Значение формулы:
\begin{itemize}
  \item $\llbracket \operatorname{Pr}(t_1,\dots,t_n)\rrbracket = P(\operatorname{Pr})(\llbracket t_1\rrbracket,\dots,\llbracket t_n\rrbracket)$;
  \item для связок — по таблицам истинности;
  \item $\llbracket \forall x.\varphi\rrbracket = \text{И}$ iff для всех $d\in D$ $\llbracket\varphi\rrbracket^{x:=d}=\text{И}$;
  \item $\llbracket \exists x.\varphi\rrbracket = \text{И}$ iff существует $d\in D$ такой, что $\llbracket\varphi\rrbracket^{x:=d}=\text{И}$.
\end{itemize}
Если $\llbracket\varphi\rrbracket=\text{И}$ в $\mathcal{M}$ при оценке $E$, пишем $\mathcal{M}\models\varphi$. Формула \textit{общезначима} ($\models\varphi$), если истинна во всех моделях.

\section{Нормальные формы}
\subsection{Предварённая нормальная форма}
Формула имеет \textit{предварённую нормальную форму}, если все кванторы вынесены в начало: 
$Q_1x_1\dots Q_nx_n.\psi$, где $\psi$ бескванторна.
\begin{theorem}
Для любой замкнутой формулы $\varphi$ существует формула $\varphi'$ в предварённой форме, эквивалентная $\varphi$ (т.е. $\vdash\varphi\leftrightarrow\varphi'$).
\end{theorem}
\textit{Доказательство (идея).} Индукция по построению $\varphi$ с использованием законов перемещения кванторов: $\neg\forall x.\alpha \equiv \exists x.\neg\alpha$, $\neg\exists x.\alpha \equiv \forall x.\neg\alpha$, а также вынесение кванторов через $\vee$ и $\wedge$ при условии переименования связанных переменных. \hfill $\square$

\subsection{Сколемизация}
Чтобы избавиться от кванторов существования, их заменяют функциями Сколема. Пусть $\varphi$ замкнута и имеет вид $\forall x_1\exists y_1\dots\forall x_n\exists y_n.\psi$ (кванторы чередуются). Тогда \textit{сколемизация} даёт формулу
$\forall x_1\dots\forall x_n.\psi[y_1:=f_1(x_1),\,y_2:=f_2(x_1,x_3),\dots]$, где $f_i$ — новые функциональные символы.
\begin{theorem}[Сколема]
Формула $\varphi$ выполнима тогда и только тогда, когда её сколемизированная форма $\eta$ выполнима. (Общезначимость не сохраняется.)
\end{theorem}
\textit{Доказательство.} $(\Rightarrow)$ Если $\varphi$ истинна в $\mathcal{M}$, то для каждого $x_1$ существует $y_1$, и т.д. Определим $f_1$ как функцию, выбирающую такой $y_1$; аналогично для остальных. В расширенной модели с такими функциями $\eta$ будет истинна. $(\Leftarrow)$ Если $\eta$ истинна в $\mathcal{M}$ (с интерпретацией $f_i$), то в той же области $\varphi$ истинна, поскольку существование $y_i$ обеспечивается функциями $f_i$. \hfill $\square$

\subsection{КНФ}
После удаления кванторов существования (сколемизации) и вынесения кванторов всеобщности, формулу можно привести к \textit{конъюнктивной нормальной форме}:
$\forall x_1\dots\forall x_k.\bigwedge_{i=1}^m \bigvee_{j=1}^{n_i} L_{ij}$,
где $L_{ij}$ — литералы (атомы или их отрицания). Далее кванторы всеобщности часто опускают, а формулу представляют множеством дизъюнктов (клаузов). Заметим, что кванторы можно внести внутрь конъюнкции перед каждым дизъюнктом, то есть выбор переменных в дизъюнктах может осуществляться независимо.

\section{Эрбрановские интерпретации}
Пусть $S$ — множество дизъюнктов (без кванторов). 
\textit{Эрбранов универсум} $H_S$ строится индуктивно:
\begin{enumerate}
  \item $H_S^0$ — множество всех констант (нульместных функциональных символов), встречающихся в $S$; если констант нет, добавляется одна (например, $a$), чтобы предметное множество было непустое;
  \item $H_S^{n+1}=H_S^n\cup\{f(t_1,\dots,t_k)\mid f$ — функциональный символ из $S$, $t_i\in H_S^n\}$.
\end{enumerate}
$H_S=\bigcup_n H_S^n$ — множество всех \textit{основных термов}.
\textit{Эрбранова интерпретация} $\mathcal{M}_H$:
\begin{itemize}
  \item предметная область $D=H_S$;
  \item функциональные символы интерпретируются синтаксически: $f(t_1,\dots,t_k)$ — сам терм;
  \item предикатные символы интерпретируются произвольным образом (задаётся множество истинных основных атомов);
  \item интерпретация переменных $E$ не задаётся, поскольку рассматриваются только замкнутые формулы (все переменные явно или неявно связаны кванторами всеобщности).
\end{itemize}
Идея в том, что эрбрановы интерпретации позволяют перебирать подстановки переменных в формуле $S$ из множества <<констант>> (эрбранова универсума), не погружаясь во внутреннюю структуру предметной области моделей.
\begin{theorem}[о существовании модели в эрбрановом универсуме]
Множество дизъюнктов $S$ выполнимо тогда и только тогда, когда оно истинно в некоторой эрбрановой интерпретации.
\end{theorem}
\textit{Доказательство.} $(\Leftarrow)$ Очевидно: эрбранова интерпретация — частный случай модели.  
$(\Rightarrow)$ Пусть $\mathcal{M}=\langle D_\mathcal{M},F_\mathcal{M},P_\mathcal{M}\rangle$ — модель $S$. Построим эрбранову интерпретацию $\mathcal{M}_H$:
\begin{enumerate}
  \item Область $D_H=H_S$ (основные термы).
  \item Каждому $n$-местному функциональному символу $f$ сопоставим синтаксическую функцию: $$F_H(f)(t_1 \in H_S,\dots,t_n)=f(t_1,\dots,t_n) \in H_S$$ (строка).
  \item Определим функцию семантики $\operatorname{eval} : H_S \rightarrow D_\mathcal{M}$, отображающую термы в исходную предметную область модели $\mathcal{M}$. Определяем индуктивно: $$\operatorname{eval}(f(t_1 \in H_S,\dots,t_k))=F_\mathcal{M}(f)(\operatorname{eval}(t_1) \in D_\mathcal{M},\dots,\operatorname{eval}(t_k)) \in D_\mathcal{M}$$
  \item Предикаты определим так: положим $$P_H(\operatorname{Pr})(t_1,\dots,t_n)=P_\mathcal{M}(\operatorname{Pr})(\operatorname{eval}(t_1),\dots,\operatorname{eval}(t_n)) \in \{\text{И}, \text{Л}\}$$
\end{enumerate}
Проверим, что $\mathcal{M}_H\models S$. Для любого дизъюнкта $\delta(x_1,\dots,x_k)\in S$ и любой подстановки $\theta$ основных термов вместо переменных, $\delta[\theta]$ истинен в $\mathcal{M}_H$ по построению предикатов. Следовательно, $\mathcal{M}_H\models\forall x_1\dots\forall x_k.\delta$ ($x_1, \dots, x_k$ перебираются из основных термов $H_S$, а не из предметной области $D_\mathcal{M}$!), а значит $\mathcal{M}_H\models S$. \hfill $\square$

\textit{Основной пример} дизъюнкта $\delta$ — результат замены всех переменных на основные термы из $H_S$. Множество всех основных примеров дизъюнктов из $S$ обозначим $G(S)$.

Будем говорить, что множество основных примеров дизъюнктов $G(S)$ выполнимо в пропозициональном смысле, если оно выполнимо при рассмотрении основных атомов $\{P(t_1,\dots,t_n)\}$ как уникальных пропозициональных переменных (без погружения во внутреннюю структуру). Заметим, что выполнимость $G(S)$ в пропозициональном смысле эквивалентна выполнимости $G(S)$ в эрбрановских интерпретациях, поскольку в них задаётся множество истинных основных атомов.
\begin{theorem}[Эрбрана]
Множество дизъюнктов $S$ невыполнимо тогда и только тогда, когда существует конечное невыполнимое в пропозициональном смысле подмножество $G(S)$.
\end{theorem}
\textit{Доказательство.}
$(\Leftarrow)$ Если конечное подмножество $G(S)$ невыполнимо (в пропозициональном смысле), то никакая эрбранова интерпретация не может удовлетворить все эти основные примеры, следовательно $S$ невыполнимо ни в какой модели (по предыдущей теореме).
$(\Rightarrow)$ Пусть $S$ невыполнимо. Тогда по теореме о существовании модели в эрбрановом универсуме $S$ невыполнимо и в эрбрановых интерпретациях. Невыполнимость $S$ в эрбрановых интерпретациях означает, что система дизъюнктов $G(S)$ невыполнима в пропозициональном смысле. По теореме компактности для пропозициональной логики (каждое конечное подмножество выполнимо $\Rightarrow$ всё множество выполнимо), из невыполнимости $G(S)$ следует существование конечного невыполнимого подмножества. \hfill $\square$

\section{Неразрешимость}
\begin{theorem}
Не существует алгоритма, который по произвольной формуле логики предикатов определяет, является ли она общезначимой (или выполнимой). Иными словами, проблема общезначимости и выполнимости неразрешима.
\end{theorem}
\textit{Доказательство (редукция к проблеме останова).} Зафиксируем машину Тьюринга $M$ и входное слово $w$. Построим замкнутую формулу $\varphi_{M,w}$ такую, что $\varphi_{M,w}$ выполнима тогда и только тогда, когда $M$ не останавливается на $w$. Используем язык с двумя функциональными символами: константой $\varepsilon$ (пустая ячейка) и бинарным функтором $c$ для построения списков (ленты). Предикат $F(w_l,w_r,s)$ означает, что конфигурация с левой частью ленты $rev(w_l)$, правой $w_r$ и состоянием $s$ достижима. Начальная конфигурация: $F(\varepsilon,w,s_0)$. Для каждого перехода записываются аксиомы, например, для перехода $(q,a)\to(q',b,R)$:
\[
\forall w_l\forall w_r\;F(w_l,c(a,w_r),q)\;\to\;F(c(b,w_l),w_r,q').
\]
Для левых переходов аналогично. Добавляем аксиому, что достижимость допускающего состояния означает остановку: $\exists w_l\exists w_r\,F(w_l,w_r,q_f)$. Пусть $C$ — конъюнкция всех таких аксиом (включая начальную). Тогда $M$ останавливается на $w$ тогда и только тогда, когда $C\models\exists w_l\exists w_r\,F(w_l,w_r,q_f)$. Это эквивалентно невыполнимости формулы $C\wedge\neg\exists w_l\exists w_r\,F(w_l,w_r,q_f)$. Таким образом, если бы существовал алгоритм, решающий выполнимость, то он бы решал и проблему останова, что невозможно. Следовательно, выполнимость (а значит, и общезначимость, т.к. $\varphi$ общезначима $\iff$ $\neg\varphi$ невыполнима) неразрешима. \hfill $\square$

\section{Метод резолюции в логике предикатов}
\subsection{Унификация}
\textit{Унификатор} для двух атомарных формул (или термов) $\alpha$ и $\beta$ — подстановка $\sigma$, такая что $\sigma(\alpha)=\sigma(\beta)$. \textit{Наиболее общий унификатор} (НОУ) $\sigma$ — унификатор, через который пропускается любой другой унификатор (т.е. для любого унификатора $\tau$ существует $\lambda$: $\tau=\lambda\circ\sigma$).
\begin{example}
$\mathcal{U}[P(x,g(b)), P(f(a),y)] = \{x\mapsto f(a),\, y\mapsto g(b)\}$.
\end{example}
\subsection{Правило резолюции}
Пусть $C_1 = L_1\vee\dots\vee L_m$, $C_2 = M_1\vee\dots\vee M_n$ — дизъюнкты, не имеющие общих переменных. Если существует литерал $L_i$ и литерал $\neg M_j$, унифицируемые с НОУ $\sigma$, то \textit{резольвентой} называется дизъюнкт
\[
\sigma(C_1\setminus\{L_i\})\vee\sigma(C_2\setminus\{M_j\}).
\]
\begin{theorem}[Полнота метода резолюции]
Множество дизъюнктов $S$ невыполнимо тогда и только тогда, когда из $S$ можно вывести пустой дизъюнкт $\Box$ с помощью правила резолюции (с учётом унификации).
\end{theorem}
\textit{Идея доказательства.} Используется теорема Эрбрана: $S$ невыполнимо $\iff$ существует конечное невыполнимое множество основных примеров $G(S)$. В пропозициональной резолюции для этого конечного множества существует вывод $\Box$. Каждый шаг пропозициональной резолюции можно поднять до шага резолюции первого порядка с использованием унификации. Подробное доказательство основано на лемме о подъёме (lifting lemma). \hfill $\square$

\begin{remark}
Требование отсутствия общих переменных исходит из независимости перебора значений переменных в разных дизъюнктах~--- мы всегда можем переименовать переменные, получив эквивалентную формулу. Тут можно сравнить с порождением $G(S)$: основные примеры дизъюнктов строятся независимо друг от друга~--- этот переход использовался в доказательстве теоремы Эрбрана.
\end{remark}
\begin{example}
    Рассмотрим формулу $$\forall x. P(x) \wedge \neg P(f(x))$$
    Не существует унификатора $f(x)$ и $x$ (<<occurs check>>~--- рекурсия), и следовательно не существует вывода пустого дизъюнкта с помощью правила резолюции \textbf{без замены переменных}. Однако формула невыполнима, так как эквивалентна формуле $(\forall x. P(x)) \wedge (\forall x. \neg P(f(x)))$, а значит замена переменных необходима для полноты метода резолюции.
\end{example}

\subsection{Процедура резолюции}
Для проверки общезначимости $\varphi$:
\begin{enumerate}
  \item строим $\neg\varphi$;
  \item приводим $\neg\varphi$ к предварённой форме, сколемизируем, затем к КНФ и получаем множество дизъюнктов $S$;
  \item применяем правило резолюции к дизъюнктам из $S$ и полученным резольвентам;
  \item если получаем $\Box$, то $\neg\varphi$ невыполнима, значит $\varphi$ общезначима;
  \item процесс может не остановиться, если $\varphi$ не общезначима.
\end{enumerate}
В отличие от пропозиционального случая, резолюция в логике предикатов — полуразрешимая процедура.

\section{Применение метода резолюции}

Известно, что:
\begin{itemize}
    \item Даша любит Сашу.
    \item Саша любит пиво.
    \item Паша любит пиво и всех тех, кто любит то, что любит Паша.
\end{itemize}
Вопрос: кто любит Дашу?

\subsection*{Формализация на языке логики предикатов}
Алфавит:
\begin{itemize}
    \item Константы: Даша, Саша, Паша, пиво.
    \item Предикатный символ $L^{(2)}$: $L(x, y)$ означает ``$x$ любит $y$''.
\end{itemize}
Формулы:
\begin{align*}
    \varphi_1 &: L(\text{Даша}, \text{Саша}), \\
    \varphi_2 &: L(\text{Саша}, \text{пиво}), \\
    \varphi_3 &: L(\text{Паша}, \text{пиво}), \\
    \varphi_4 &: \forall x \bigl( \exists y\, (L(\text{Паша}, y) \land L(x, y)) \to L(\text{Паша}, x) \bigr), \\
    \varphi_0 &: \exists z\, L(z, \text{Даша}) \quad\text{(вопрос)}.
\end{align*}
Нужно проверить, верно ли $\{\varphi_1,\varphi_2,\varphi_3,\varphi_4\} \models \varphi_0$.

\subsection*{Приведение к опровержению}
Проверяем общезначимость импликации:
\[
\models \bigl(\varphi_1 \land \varphi_2 \land \varphi_3 \land \varphi_4\bigr) \to \varphi_0.
\]
Отрицание заключения:
\[
\neg\varphi_0 = \forall z\, \neg L(z, \text{Даша}).
\]
Получаем множество дизъюнктов после приведения к предваренной нормальной форме и сколемизации:
\begin{align*}
    S = \{
    & D_1 = L(\text{Даша}, \text{Саша}), \\
    & D_2 = L(\text{Саша}, \text{пиво}), \\
    & D_3 = L(\text{Паша}, \text{пиво}), \\
    & D_4 = \neg L(\text{Паша}, y) \lor \neg L(x, y) \lor L(\text{Паша}, x), \\
    & D_0 = \neg L(z, \text{Даша}) \}.
\end{align*}

\subsection*{Линейный резолютивный вывод}
Строим вывод, начиная с дизъюнкта-запроса $D_0$.

\subsubsection*{Шаг 1}
\[
\begin{array}{ll}
D_0 = \neg L(z, \text{Даша}), & D_4 = \neg L(\text{Паша}, y_1) \lor \neg L(x_1, y_1) \lor L(\text{Паша}, x_1), \\
\theta_1 = \{z / \text{Паша},\; x_1 / \text{Даша}\}, & \\
D_1' = \neg L(\text{Паша}, y_1) \lor \neg L(\text{Даша}, y_1). &
\end{array}
\]

\subsubsection*{Шаг 2}
\[
\begin{array}{ll}
D_1' = \neg L(\text{Паша}, y_1) \lor \neg L(\text{Даша}, y_1), & D_1 = L(\text{Даша}, \text{Саша}), \\
\theta_2 = \{y_1 / \text{Саша}\}, & \\
D_2' = \neg L(\text{Паша}, \text{Саша}). &
\end{array}
\]

\subsubsection*{Шаг 3}
\[
\begin{array}{ll}
D_2' = \neg L(\text{Паша}, \text{Саша}), & D_4 = \neg L(\text{Паша}, y_2) \lor \neg L(x_2, y_2) \lor L(\text{Паша}, x_2), \\
\theta_3 = \{x_2 / \text{Саша}\}, & \\
D_3' = \neg L(\text{Паша}, y_2) \lor \neg L(\text{Саша}, y_2). &
\end{array}
\]

\subsubsection*{Шаг 4}
\[
\begin{array}{ll}
D_3' = \neg L(\text{Паша}, y_2) \lor \neg L(\text{Саша}, y_2), & D_2 = L(\text{Саша}, \text{пиво}), \\
\theta_4 = \{y_2 / \text{пиво}\}, & \\
D_4' = \neg L(\text{Паша}, \text{пиво}). &
\end{array}
\]

\subsubsection*{Шаг 5}
\[
\begin{array}{ll}
D_4' = \neg L(\text{Паша}, \text{пиво}), & D_3 = L(\text{Паша}, \text{пиво}), \\
\theta_5 = \varepsilon, & \\
D_5' = \square. &
\end{array}
\]

Пустой дизъюнкт получен, следовательно, исходное множество дизъюнктов противоречиво, а формула $\varphi_0$ является логическим следствием посылок.

\subsection*{Нахождение ответа}
Проследим, какое значение получила переменная $z$ в дизъюнкте-запросе $D_0 = \neg L(z, \text{Даша})$ после применения всех унификаторов:
\[
\theta_1 = \{z/\text{Паша},\; x_1/\text{Даша}\},\;
\theta_2 = \{y_1/\text{Саша}\},\;
\theta_3 = \{x_2/\text{Саша}\},\;
\theta_4 = \{y_2/\text{пиво}\},\;
\theta_5 = \varepsilon.
\]
\[
z\theta_1\theta_2\theta_3\theta_4\theta_5 = \text{Паша}.
\]
\textbf{Ответ:} Паша любит Дашу.

\section*{Заключение}
Логика предикатов первого порядка обладает богатой выразительностью, но неразрешима. Метод резолюции, основанный на унификации и теореме Эрбрана, позволяет автоматически доказывать общезначимые формулы, хотя и не гарантирует остановки для необщезначимых. Ключевые понятия: нормальные формы, эрбрановские интерпретации, унификация — составляют основу автоматического доказательства теорем.
