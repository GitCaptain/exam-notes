\section{Статистическая устойчивость среднего}

При многократном повторении независимых экспериментов среднее арифметическое наблюдений стремится к теоретическому среднему (математическому ожиданию). Это свойство называется \textbf{статистической устойчивостью} и лежит в основе законов больших чисел.

\section{Сходимость последовательностей случайных величин}

\begin{theorem}[Неравенство Маркова]
Для неотрицательной случайной величины $X$ и любого $a > 0$:
\[
\Pr(X \ge a) \le \frac{E[X]}{a}.
\]
\end{theorem}
\begin{proof}
Определим вспомогательную случайную величину $Y = \begin{cases} a, & X \ge a, \\ 0, & X < a. \end{cases}$ Тогда $Y \le X$, следовательно $E[Y] \le E[X]$. Но $E[Y] = a \Pr(X \ge a)$. Отсюда $\Pr(X \ge a) = \frac{E[Y]}{a} \le \frac{E[X]}{a}$.
\end{proof}

\begin{theorem}[Неравенство Чебышёва]
Для любой случайной величины $X$ с конечной дисперсией $\sigma^2 = \operatorname{Var}(X)$ и любого $\varepsilon > 0$:
\[
\Pr(|X - E[X]| \ge \varepsilon) \le \frac{\sigma^2}{\varepsilon^2}.
\]
\end{theorem}

\begin{proof}
\[
\Pr(|X - \mu| \ge \varepsilon) = \Pr((X-\mu)^2 \ge \varepsilon^2) \le \frac{E[(X-\mu)^2]}{\varepsilon^2} = \frac{\sigma^2}{\varepsilon^2},
\]
где использовано неравенство Маркова для неотрицательной с.в. $(X-\mu)^2$.
\end{proof}

Пусть $\{X_n\}_{n\ge 1}$ и $X$ — случайные величины, определённые на одном вероятностном пространстве.

\begin{definition}[Сходимость по вероятности]
$X_n \xrightarrow{\text{п.в.}} X$, если для любого $\varepsilon > 0$:
\[
\lim_{n\to\infty} \Pr(|X_n - X| \ge \varepsilon) = 0.
\]
\end{definition}

\begin{definition}[Сходимость почти наверное]
$X_n \xrightarrow{\text{п.н.}} X$, если $\Pr(\lim_{n\to\infty} X_n = X) = 1$. (то есть сходится везде, кроме множества нулевой вероятности).
\end{definition}

\begin{definition}[Сходимость по распределению]
$X_n \xrightarrow{\text{п.р.}} X$, если функции распределения $F_{X_n}(x)$ сходятся к $F_X(x)$ во всех точках непрерывности $F_X$.
\end{definition}

\begin{definition}[Сходимость в среднем (в $L^p$)]
$X_n \xrightarrow{L^p} X$, если $\lim_{n\to\infty} E[|X_n - X|^p] = 0$ (для $p \ge 1$). Чаще всего рассматривают $p=2$ (сходимость в среднеквадратическом).
\end{definition}

\subsection*{Связь между видами сходимости}
\begin{theorem}
Справедливы следующие импликации:
\begin{enumerate}
    \item $X_n \xrightarrow{\text{п.н.}} X \;\Rightarrow\; X_n \xrightarrow{\text{п.в.}} X$.
    \item $X_n \xrightarrow{L^p} X \;\Rightarrow\; X_n \xrightarrow{\text{п.в.}} X$.
    \item $X_n \xrightarrow{\text{п.в.}} X \;\Rightarrow\; X_n \xrightarrow{\text{п.р.}} X$.
\end{enumerate}
Обратные утверждения, вообще говоря, неверны.
\end{theorem}

\begin{proof}[Обоснование]
\begin{enumerate}
    \item Если $X_n \xrightarrow{\text{п.н.}} X$, то для любого $\varepsilon>0$ множество $\{\omega: \liminf_{n} |X_n(\omega)-X(\omega)| \ge \varepsilon\}$ имеет вероятность $0$. Отсюда $\Pr(|X_n-X| \ge \varepsilon) \to 0$ (это следует из определения сходимости почти наверное).
    \item Из неравенства Маркова для неотрицательной с.в. $|X_n-X|^p$:
    \[
    \Pr(|X_n-X| \ge \varepsilon) = \Pr(|X_n-X|^p \ge \varepsilon^p) \le \frac{E[|X_n-X|^p]}{\varepsilon^p} \xrightarrow[n\to\infty]{} 0.
    \]
    \item Для любой точки непрерывности $x$ функции распределения $F_X$ и любого $\varepsilon>0$:
    \[
    F_{X_n}(x) - F_X(x) \le \Pr(X_n \le x, X > x+\varepsilon) + \Pr(X > x+\varepsilon) + \Pr(|X_n-X| \ge \varepsilon).
    \]
    Аналогично оценивается $F_X(x) - F_{X_n}(x)$. Устремляя $\varepsilon\to 0$ и используя сходимость по вероятности, получаем $F_{X_n}(x) \to F_X(x)$.
\end{enumerate}
\end{proof}

\subsection*{Контрпримеры к обратным утверждениям}
\begin{itemize}
    \item \textbf{Сходимость по вероятности не влечёт почти наверное:} Пусть $X_n = 1$ с вероятностью $1/n$, и $X_n = 0$ иначе. Тогда $\Pr(|X_n| \ge \varepsilon) = 1/n \to 0$, но $\sum \Pr(|X_n| \ge 1) = \sum 1/n = \infty$, поэтому по лемме Бореля–Кантелли с вероятностью $1$ бесконечно много $X_n$ равны $1$, и $X_n$ не сходится к $0$ почти наверное.
    \item \textbf{Сходимость по распределению не влечёт по вероятности:} Пусть $X \sim N(0,1)$ и $X_n = X$ для всех $n$. Тогда $X_n \xrightarrow{\text{п.р.}} X$ (тривиально), но $\Pr(|X_n-X| \ge 1) = 1$, поэтому сходимости по вероятности нет.
    \item \textbf{Сходимость по вероятности не влечёт сходимость в $L^p$:} Рассмотрим $X_n = n$ с вероятностью $1/n$, и $0$ иначе. Тогда $X_n \xrightarrow{\text{п.в.}} 0$, но $E[|X_n|] = 1 \not\to 0$, поэтому сходимости в $L^1$ нет.
\end{itemize}


\section{Закон больших чисел}

\subsection{Слабый закон больших чисел (форма Чебышёва)}

\begin{theorem}[Слабый закон больших чисел]
Пусть $X_1, X_2, \dots$ — независимые одинаково распределённые случайные величины с конечным математическим ожиданием $\mu$ и конечной дисперсией $\sigma^2$. Обозначим $M_n = \frac{1}{n}\sum_{i=1}^n X_i$. Тогда для любого $\varepsilon > 0$:
\[
\lim_{n\to\infty} \Pr(|M_n - \mu| \ge \varepsilon) = 0.
\]
\end{theorem}

\begin{proof}
Математическое ожидание $M_n$ равно $\mu$, а дисперсия $\operatorname{Var}(M_n) = \sigma^2/n$. Применяя неравенство Чебышёва к $M_n$:
\[
\Pr(|M_n - \mu| \ge \varepsilon) \le \frac{\sigma^2/n}{\varepsilon^2} \xrightarrow[n\to\infty]{} 0.
\]
\end{proof}

\begin{corollary}
Сходимость $\Pr(|M_n - \mu| \ge \varepsilon) \to 0$~--- \textbf{сходимость по вероятности}. Слабый закон утверждает, что $M_n \xrightarrow{\text{п.в.}} \mu$.
\end{corollary}

\subsection{Слабый закон больших чисел Хинчина}
Если $X_1, X_2, \dots$ — независимые одинаково распределённые случайные величины с конечным математическим ожиданием $\mu$ (дисперсия может не существовать), то $M_n \xrightarrow{\text{п.в.}} \mu$. Доказательство использует характеристические функции.

\subsection{Сильный закон больших чисел (Колмогорова)}
При тех же условиях более сильное утверждение:
\[
\Pr\!\left(\lim_{n\to\infty} M_n = \mu\right) = 1.
\]
Сходимость происходит \textbf{почти наверное}. Доказательство опускается.

\section{Центральная предельная теорема (ЦПТ)}

Слабый закон говорит, что среднее близко к $\mu$, но не описывает распределение отклонений. ЦПТ показывает, что нормированная сумма приближается к нормальному распределению.

\begin{theorem}[Центральная предельная теорема]
Пусть $X_1, X_2, \dots$ — независимые одинаково распределённые случайные величины с конечным математическим ожиданием $\mu$ и конечной дисперсией $\sigma^2 > 0$. Определим
\[
S_n = \sum_{i=1}^n X_i, \qquad Z_n = \frac{S_n - n\mu}{\sigma\sqrt{n}}.
\]
Тогда для любого $x \in \mathbb{R}$:
\[
\lim_{n\to\infty} \Pr(Z_n \le x) = \Phi(x) = \int_{-\infty}^x \frac{1}{\sqrt{2\pi}} e^{-t^2/2}\,dt,
\]
т.е. $Z_n$ сходится по распределению к стандартному нормальному закону $N(0,1)$.
\end{theorem}

\begin{example}
Вес контейнера $X_i \sim \text{Exp}(1/2)$, $\mu = 2$, $\sigma = 2$. Для $n=100$:
\[
\Pr(S_{100} > 210) = \Pr\!\left(Z_{100} > \frac{210 - 100\cdot2}{2\sqrt{100}}\right) = \Pr(Z_{100} > 0.5) \approx 1 - \Phi(0.5) \approx 0.3085.
\]
\end{example}

\begin{theorem}[Муавра–Лапласа (частный случай ЦПТ)]
Для биномиального распределения $S_n \sim \text{Bin}(n,p)$ с $p \in (0,1)$ и $q=1-p$:
\[
\frac{S_n - np}{\sqrt{npq}} \xrightarrow{\text{п.р.}} N(0,1).
\]
\end{theorem}

\subsection{Скорость сходимости: теорема Берри–Эссена}
Если $E[|X_i|^3] = \rho < \infty$, то существует константа $C>0$ (например, $C=0.33$) такая, что для всех $n$ и всех $x$:
\[
\bigl|\Pr(Z_n \le x) - \Phi(x)\bigr| \le \frac{C\rho}{\sigma^3\sqrt{n}}.
\]
Это означает, что погрешность аппроксимации убывает как $O(1/\sqrt{n})$.

\section{Практическое применение ЦПТ}
ЦПТ позволяет приближать распределение суммы больших выборок нормальным законом, что широко используется в статистике, финансовом моделировании и многих других областях. Для вычисления вероятностей используют таблицы функции $\Phi(x)$.

\section{Локальная лемма Ловаса}
Локальная лемма Ловаса (LLL) — мощный вероятностный метод для доказательства существования объектов, удовлетворяющих набору «плохих» событий, каждое из которых зависит лишь от небольшого числа других.

\begin{definition}
Пусть $\mathcal{E} = \{E_1, \dots, E_n\}$ — конечный набор событий в вероятностном пространстве. Для каждого $E_i$ задано множество индексов $\Gamma(i) \subset \{1,\dots,n\} \setminus \{i\}$ такое, что $E_i$ не зависит от всех событий $E_j$ с $j \notin \Gamma(i) \cup \{i\}$ (зависимость только от событий с индексами из $\Gamma(i)$).
\end{definition}

\begin{theorem}[Локальная лемма Ловаса, симметричная версия]
Если для всех $i$ выполнено $\Pr(E_i) \le p$ и каждый $E_i$ зависит не более чем от $d$ других событий, причём $ep(d+1) \le 1$ (где $e \approx 2.718$), то $\Pr(\bigcap_i \overline{E_i}) > 0$, т.е. существует исход, не принадлежащий ни одному из $E_i$.
\end{theorem}

\begin{example}[Раскраска графа]
Пусть граф $G$ имеет максимальную степень $\Delta$ и каждое ребро раскрашивается в один из $k$ цветов. Если $k \ge 2e\Delta$, то существует правильная раскраска (соседние рёбра разного цвета). Доказательство использует LLL: за плохие события берутся случаи, когда два смежных ребра получают одинаковый цвет.
\end{example}
