\section{Случайные процессы: основные понятия}

\begin{definition}
Условная вероятность события $A$ при условии, что произошло событие $B$ (при $\Pr(B)>0$), определяется как
\[
\Pr(A \mid B) = \frac{\Pr(A \cap B)}{\Pr(B)}.
\]
\end{definition}

Для дискретных случайных величин $X$ и $Y$ условная функция вероятности $X$ при условии $Y=y$ задаётся как
\[
p_{X\mid Y}(x \mid y) = \frac{p_{X,Y}(x,y)}{p_Y(y)} \quad \text{при } p_Y(y)>0.
\]
Аналогично, для непрерывных с.в. условная плотность вероятности
\[
f_{X\mid Y}(x \mid y) = \frac{f_{X,Y}(x,y)}{f_Y(y)} \quad \text{при } f_Y(y)>0.
\]

\begin{theorem}[Формула полной вероятности]
Пусть $\{A_i\}$ — конечное или счётное разбиение пространства элементарных исходов $\Omega$, т.е. $A_i \cap A_j = \varnothing$ для $i\ne j$ и $\bigcup_i A_i = \Omega$, причём $\Pr(A_i)>0$. Тогда для любого события $B$
\[
\Pr(B) = \sum_i \Pr(A_i) \Pr(B \mid A_i).
\]
Для случайных величин формула полной вероятности для функции вероятности $p_X(x)$ имеет вид
\[
p_X(x) = \sum_i \Pr(A_i) p_{X\mid A_i}(x),
\]
а для плотности
\[
f_X(x) = \sum_i \Pr(A_i) f_{X\mid A_i}(x).
\]
\end{theorem}

\subsection*{Формула полного математического ожидания}
Если $\{A_i\}$ — разбиение $\Omega$, то
\[
E[X] = \sum_i \Pr(A_i) E[X \mid A_i].
\]
Для дискретной с.в. $Y$:
\[
E[X] = \sum_y p_Y(y) E[X \mid Y=y],
\]
а для непрерывной:
\[
E[X] = \int_{-\infty}^{+\infty} f_Y(y) E[X \mid Y=y]\,dy.
\]

\begin{definition}
\textbf{Случайный процесс} (стохастический процесс) — это семейство случайных величин $\{X_t\}_{t \in T}$, определённых на одном вероятностном пространстве, где $T$ — множество индексов (обычно интерпретируемое как время). Если $T$ дискретно (например, $T = \mathbb{N}_0$), процесс называют \textbf{дискретным по времени}, если $T$ непрерывно (например, $T = \mathbb{R}_+$) — \textbf{непрерывным по времени}.
\end{definition}
\begin{definition}

Пусть $\{X_t\}_{t\ge 0}$ — случайный процесс с вещественным временем. Он называется \textbf{марковским процессом}, если выполнены следующие три условия:

\begin{enumerate}
    \item \textbf{Однородность во времени:} для всех $t, \delta, h, i, j$
    \[
    \Pr[X_{t+\delta}=j \mid X_t=i] = \Pr[X_{t+h+\delta}=j \mid X_{t+h}=i],
    \]
    или, в терминах плотностей (для непрерывных состояний),
    \[
    f_{X_{t+\delta}\mid X_t}(j\mid i) = f_{X_{t+h+\delta}\mid X_{t+h}}(j\mid i).
    \]
    То есть вероятность перехода из состояния $i$ в состояние $j$ за время $\delta$ не зависит от начального момента времени $t$, а только от самого интервала $\delta$.

    \item \textbf{Марковское свойство:} для всех $t, \delta, i, j$ и любых наборов состояний $\{i_\tau\}_{\tau<t}$ выполняется
    \[
    \Pr\bigl[X_{t+\delta}=j \mid X_t=i,\; \bigcap_{\tau<t} X_\tau=i_\tau\bigr] = \Pr[X_{t+\delta}=j \mid X_t=i].
    \]
    (и аналогично для плотностей вероятностей). Это означает, что будущее процесса зависит только от текущего состояния, а не от всей предыстории.

    \item \textbf{Условие нормировки:} для любого фиксированного $i$ и $\delta$ сумма (или интеграл) вероятностей перехода по всем возможным состояниям равна единице:
    \[
    \sum_j \Pr[X_{t+\delta}=j \mid X_t=i] = 1,
    \]
    или для непрерывного пространства состояний
    \[
    \int_{\mathbb{R}} f_{X_{t+\delta}\mid X_t}(x\mid i)\,dx = 1.
    \]
\end{enumerate}

Для марковских процессов с \textbf{дискретным временем} и \textbf{конечным числом состояний} эти условия упрощаются: достаточно рассматривать $\delta=1$ и обозначить
\[
p_{ij} = \Pr[X_{n+1}=j \mid X_n=i].
\]
Тогда условия принимают вид:
\begin{enumerate}
    \item $p_{ij}$ не зависит от $n$ (однородность);
    \item $\Pr[X_{n+1}=j \mid X_n=i, X_{n-1},\dots,X_0] = p_{ij}$ (марковское свойство);
    \item $\sum_{j} p_{ij} = 1$ для всех $i$ (нормировка).
\end{enumerate}
\end{definition}

\section{Конечные цепи Маркова}

\subsection{Определение и матрица переходов}
Пусть пространство состояний $S = \{1,2,\dots,m\}$. В силу однородности определим \textbf{вероятности перехода за один шаг}:
\[
p_{ij} = \Pr(X_{n+1} = j \mid X_n = i), \quad i,j \in S.
\]
Матрица $P = (p_{ij})_{i,j=1}^m$ называется \textbf{матрицей переходных вероятностей}. Она стохастическая (определение):
\[
p_{ij} \ge 0, \quad \sum_{j=1}^m p_{ij} = 1 \ \text{для всех } i.
\]

\subsection{Графическое представление}
Цепь Маркова удобно изображать в виде ориентированного графа, вершины которого — состояния, а рёбра $(i,j)$ присутствуют, если $p_{ij} > 0$, и снабжены весами $p_{ij}$.

\begin{example}[Очередь в супермаркете]
Пусть появление новых покупателей описывается процессом Бернулли с вероятностью $p$, а время обслуживания каждого покупателя — геометрическим распределением с параметром $q$. Если в очереди уже 10 покупателей, новые не встают. Тогда в каждый момент времени возможны события:
\begin{enumerate}
    \item Пришёл новый покупатель, обслуживание не закончилось: $p(1-q)$,
    \item Обслуживание закончилось, новый не пришёл: $q(1-p)$,
    \item Пришёл и закончилось: $pq$,
    \item Ничего не произошло: $(1-p)(1-q)$.
\end{enumerate}
Цепь Маркова для числа покупателей в очереди имеет вид:

\subsection{$n$-шаговые вероятности перехода}
Обозначим $p_{ij}^{(n)} = \Pr(X_{k+n} = j \mid X_k = i)$. Эти вероятности удовлетворяют \textbf{уравнению Чепмена–Колмогорова}.

\begin{theorem}[Уравнение Чепмена–Колмогорова]
Для любых $n,m \ge 0$ и любых $i,j \in S$:
\[
p_{ij}^{(n+m)} = \sum_{k \in S} p_{ik}^{(n)} p_{kj}^{(m)}.
\]
В матричной форме: $P^{(n+m)} = P^{(n)} P^{(m)}$, где $P^{(n)} = (p_{ij}^{(n)})$. В частности, $P^{(n)} = P^n$ (матрица $P$ в степени $n$).
\end{theorem}
\begin{proof}
Используем формулу полной вероятности, разбивая по состоянию $X_n$ (промежуточному состоянию):
\begin{align*}
   p_{ij}^{(n+m)} &= \Pr(X_{n+m}=j \mid X_0=i)\\
   &= \sum_{k} \Pr(X_n=k \mid X_0=i) \Pr(X_{n+m}=j \mid X_n=k) \\
   &= \sum_{k} p_{ik}^{(n)} p_{kj}^{(m)}.
\end{align*}
По индукции $P^{(n)} = P^n$.
\end{proof}

\subsection{Классификация состояний}
\begin{itemize}
    \item Состояние $i$ называется \textbf{возвратным}, если вероятность когда-либо вернуться в него, стартуя из $i$, равна $1$; иначе — \textbf{невозвратным}.
    \item Состояния $i$ и $j$ называются \textbf{сообщающимися}, если существуют $n,m \ge 0$ такие, что $p_{ij}^{(n)} > 0$ и $p_{ji}^{(m)} > 0$. Классы сообщающихся состояний образуют разбиение множества состояний.
    \item Цепь называется \textbf{неразложимой} (эргодической), если все состояния сообщаются (один класс).
    \item Состояние $i$ имеет \textbf{период} $d = \gcd\{ n \ge 1 : p_{ii}^{(n)} > 0\}$. Если $d=1$, состояние называется \textbf{апериодическим}. Если цепь неразложима и апериодична, она называется \textbf{апериодической неразложимой}.
\end{itemize}

\subsection{Стационарное распределение}

\begin{definition}
Распределение вероятностей $\pi = (\pi_1,\dots,\pi_m)$ (т.е. $\pi_i \ge 0$, $\sum_i \pi_i = 1$) называется \textbf{стационарным} для цепи Маркова с матрицей $P$, если
\[
\pi = \pi P, \quad \text{т.е.} \quad \pi_j = \sum_{i=1}^m \pi_i p_{ij} \ \text{для всех } j.
\]
\end{definition}

Стационарное распределение означает, что если начальное распределение равно $\pi$, то распределение в любой следующий момент также равно $\pi$.

\subsection{Эргодическая теорема для конечных цепей Маркова}

\begin{theorem}[Эргодическая теорема]
Пусть $\{X_n\}$ — конечная, неразложимая, апериодическая цепь Маркова с матрицей переходов $P$. Тогда существует единственное стационарное распределение $\pi$, и для любых начальных условий выполняется:
\[
\lim_{n \to \infty} p_{ij}^{(n)} = \pi_j \quad \text{для всех } i,j.
\]
Кроме того, $\pi$ — единственное решение системы $\pi = \pi P$ (следует из предельного перехода при $n\rightarrow \infty$ в равенстве $p_{ij}^n = \sum_{k=1}^m p_{ik}^{n-1}p_{ij}$: $\pi_j = \sum_{k=1}^m \pi_k p_{kj}$), $\sum_i \pi_i = 1$.
\end{theorem}

\textit{Идея доказательства:} Доказательство существования опускается. 

\subsection{Интерпретация предельных вероятностей}
Из эргодической теоремы следует, что для достаточно больших $n$ вероятность находиться в состоянии $j$ приблизительно равна $\pi_j$ независимо от начального состояния. Более того, для неразложимой апериодической цепи справедлив закон больших чисел: доля времени, проведённого в состоянии $j$, сходится к $\pi_j$ почти наверное. Аналогично, доля переходов $i\to j$ стремится к $\pi_i p_{ij}$.

\section{Поглощающие состояния и время до поглощения}

\subsection{Определение}
Состояние $i$ называется \textbf{поглощающим}, если $p_{ii}=1$. В цепях с несколькими поглощающими состояниями интерес представляют вероятности поглощения в каждом из них и среднее время до поглощения.

\subsection{Вероятности поглощения}
Пусть $a_i$ — вероятность попасть в поглощающее состояние $A$, стартуя из $i$. Для непоглощающих состояний из формулы полной вероятности:
\[
a_i = \sum_{j} p_{ij} a_j.
\]
Это даёт систему линейных уравнений с граничными условиями $a_j = 1$ для поглощающих состояний из $A$, $a_j = 0$ для остальных поглощающих состояний.

\begin{example}[Задача о разорении игрока]
Игрок с начальным капиталом $i$ играет в честную игру (вероятность выигрыша $1/2$). Игра заканчивается при разорении (0) или достижении капитала $n$. Вероятность $a_i$ выиграть $n$ удовлетворяет:
\[
a_0 = 0,\ a_n = 1,\ a_i = 0.5 a_{i-1} + 0.5 a_{i+1}.
\]
Решение: $a_i = i/n$.
\end{example}

\subsection{Среднее время до поглощения}
Объединим все поглощающие состояния в одно. Пусть $T_i$ — среднее число шагов до поглощения, стартуя из $i$. Тогда:
\[
T_i = 1 + \sum_{j} p_{ij} T_j,
\]
где $T_j = 0$ для поглощающих состояний. Для примера с игроком это даёт $T_i = i(n-i)$.

\section{Уравнение Чепмена–Колмогорова для непрерывных цепей}
Для непрерывных по времени цепей Маркова с конечным или счётным числом состояний также можно записать уравнение Чепмена–Колмогорова. Пусть $p_{ij}(t) = \Pr(X_{s+t}=j \mid X_s=i)$ (в силу однородности). Тогда
\[
p_{ij}(s+t) = \sum_{k \in S} p_{ik}(s) p_{kj}(t), \quad s,t \ge 0.
\]
В матричной форме: $P(s+t) = P(s) P(t)$, где $P(t) = (p_{ij}(t))$.

\section{Заключение}
Цепи Маркова являются основой для моделирования широкого круга явлений: очереди, демографические процессы, случайные блуждания, экономические и финансовые модели. Эргодическая теорема гарантирует существование стационарного режима, что позволяет делать долгосрочные прогнозы.
